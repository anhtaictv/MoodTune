function buildSearchUrl(mode, offset) {
  const base = `${API}/music/search?offset=${offset}&limit=${PAGE_SIZE}`;
  return mode.kind === 'query'
    ? `${base}&q=${encodeURIComponent(mode.value)}`
    : `${base}&emotion=${mode.value}`;
}

// ── GỢI Ý THÔNG MINH ──────────────────────────────────────────────
// Lấy top N key có giá trị đếm cao nhất từ object {key:count}
function topKeys(obj, n) {
  return Object.entries(obj).sort((a,b)=>b[1]-a[1]).slice(0,n).map(e=>e[0]);
}

async function loadRecommend() {
  searchMode = null;            // recommend không phân trang
  currentOffset = 0;
  const list = document.getElementById('song-list');
  list.innerHTML = `<div class="loading-songs"><div class="spin spin-green" style="width:20px;height:20px;margin:0 auto 10px"></div><br>Đang tạo gợi ý riêng cho bạn...</div>`;
  toggleLoadMore(false);
  const artists = topKeys(likedArtists, 3).join(",");
  const tags    = topKeys(likedTags, 3).join(",");
  try {
    const res = await fetch(`${API}/recommend?liked_artists=${encodeURIComponent(artists)}&liked_tags=${encodeURIComponent(tags)}`);
    const data = await res.json();
    if (!data.tracks || data.tracks.length === 0) {
      list.innerHTML = '<div style="color:var(--muted);font-size:12px;text-align:center;padding:24px;">Chưa đủ dữ liệu gợi ý. Hãy nghe và thích vài bài trước!</div>';

// Ghi nhận hành vi: tăng đếm nghệ sĩ/tag + gửi server (fire-and-forget)
function recordBehavior(song, type) {
  if (song.artist) {
    likedArtists[song.artist] = (likedArtists[song.artist] || 0) + (type==='like'?3:1);
    localStorage.setItem("mt_artists", JSON.stringify(likedArtists));
  }
  (song.tags || []).forEach(tag => {
    likedTags[tag] = (likedTags[tag] || 0) + (type==='like'?2:1);
  });
  localStorage.setItem("mt_tags", JSON.stringify(likedTags));
  // gửi server để thống kê chung (không chặn UI)
  fetch(`${API}/track/event`, {
    method:"POST", headers:{"Content-Type":"application/json"},
    body: JSON.stringify({type, track_id:song.id, name:song.name,
                          artist:song.artist, tags:song.tags||[]})
  }).catch(()=>{});
}

function renderSongList(tracks) {
  const list = document.getElementById('song-list');
  list.innerHTML = tracks.map((t,i) => {
    const fb = userFeedback[t.id];
    const fbIcon = fb==='like'?' 👍':fb==='dislike'?' 👎':'';
    const thumb = t.image
      ? `<img class="song-thumb" src="${t.image}" alt="">`
      : `<div class="song-thumb-ph">🎵</div>`;
    return `<div class="song-item" id="si${i}" onclick="playSong(${i})">
      ${thumb}
      <div class="song-info">
        <div class="song-title">${t.name}${fbIcon}</div>
        <div class="song-artist">${t.artist}</div>
        <div class="song-album">${t.album}</div>
      </div>
      <div class="song-right">
        ${t.preview ? '<span class="preview-badge">FULL</span>' : ''}
      </div>
      <button class="song-play" id="play${i}" title="Phát / Tạm dừng" onclick="event.stopPropagation();clickSongPlay(${i})">▶</button>
    </div>`;
  }).join('');
}

// ── PLAYER ────────────────────────────────────────────────────────
// Cập nhật icon nút ▶/⏸ ở player chính và trong từng item của list
function syncPlayIcons() {
  const icon = previewPlaying ? '⏸' : '▶';
  document.getElementById('play-btn').textContent = icon;
  document.querySelectorAll('.song-play').forEach((b,i) => {
    b.textContent = (i === currentIdx) ? icon : '▶';
  });
}

function playSong(idx) {
  currentIdx = idx;
  const s = currentSongs[idx];
  document.querySelectorAll('.song-item').forEach((el,i) => el.classList.toggle('playing', i===idx));

  document.getElementById('p-title').textContent  = s.name;
  document.getElementById('p-artist').textContent = s.artist;
  document.getElementById('p-album').textContent  = s.album;

  const cover = document.getElementById('p-cover');
  const ph    = document.getElementById('p-cover-ph');
  if (s.image) { cover.src=s.image; cover.style.display='block'; ph.style.display='none'; }
  else { cover.style.display='none'; ph.style.display='flex'; }

  const audio = document.getElementById('preview-audio');
  const pp    = document.getElementById('preview-player');
  if (s.preview) {
    audio.src = s.preview;
    pp.classList.add('show');
    audio.play();
    previewPlaying = true;
    recordBehavior(s, 'play');   // ghi nhận lượt nghe để gợi ý thông minh
  } else {
    pp.classList.remove('show');
    audio.src = '';
    previewPlaying = false;
    toast("⚠️ Bài này không có file nhạc");
  }
  syncPlayIcons();

  const openBtn = document.getElementById('open-sp');
  openBtn.href = s.open_url;
  openBtn.style.display = 'flex';

  document.getElementById('fb-row').style.display = 'flex';
  document.getElementById('fb-like').className    = 'fb-btn' + (userFeedback[s.id]==='like'?' liked':'');
  document.getElementById('fb-dislike').className = 'fb-btn' + (userFeedback[s.id]==='dislike'?' disliked':'');
}

function togglePreview() {
  const audio = document.getElementById('preview-audio');
  if (!audio.src) {
    // Chưa chọn bài nào → phát bài đầu tiên trong list
    if (currentSongs.length) playSong(0);
    return;
  }
  if (previewPlaying) { audio.pause(); previewPlaying=false; }
  else { audio.play(); previewPlaying=true; }
  syncPlayIcons();
}

function stopSong() {
  const audio = document.getElementById('preview-audio');
  audio.pause();
  audio.currentTime = 0;
  previewPlaying = false;
  syncPlayIcons();
  toast("⏹ Đã dừng");
}

// Bấm nút ▶ tròn trên một bài trong list
function clickSongPlay(idx) {
  if (idx === currentIdx) togglePreview();  // đang là bài này → play/pause
  else playSong(idx);                        // bài khác → phát bài đó
}

function nextSong() { if (currentSongs.length) playSong((currentIdx+1)%currentSongs.length); }
function prevSong() { if (currentSongs.length) playSong((currentIdx-1+currentSongs.length)%currentSongs.length); }

// ── FEEDBACK ──────────────────────────────────────────────────────
function feedback(type) {
  if (currentIdx<0) return;
  const s = currentSongs[currentIdx];
  userFeedback[s.id] = type;
  localStorage.setItem("mt_fb", JSON.stringify(userFeedback));
  document.getElementById('fb-like').className    = 'fb-btn'+(type==='like'?' liked':'');
  document.getElementById('fb-dislike').className = 'fb-btn'+(type==='dislike'?' disliked':'');
  recordBehavior(s, type);   // like/dislike → cập nhật sở thích + gửi server
  toast(type==='like'?'✅ Đã lưu!':'❌ Đã ghi nhận!');
}

    const res = await fetch(API+"/learn", {
      method:"POST", headers:{"Content-Type":"application/json"},
      body: JSON.stringify({text:currentText, correct_emotion:emo})
    });
    const d = await res.json();
    statusEl.textContent = `✓ ${d.message} | Total: ${d.feedback_count} | Alpha: ${d.new_alpha.toFixed(2)}`;
    document.getElementById('stat-feedback').textContent = d.feedback_count;
    document.getElementById('stat-alpha').textContent = d.new_alpha.toFixed(2);
    toast("🧠 Model đã học xong!");
  } catch { statusEl.textContent = "⚠️ Không kết nối server."; }
}

function addHistory(text, data) {
  history.unshift({text, emoji:data.emoji, label:data.label, conf:(data.confidence*100).toFixed(0)+'%', time:new Date().toLocaleString('vi-VN')});
  if (history.length>20) history.pop();
  localStorage.setItem("mt_hist", JSON.stringify(history));
  renderHistory();
}
function renderHistory() {
  const el = document.getElementById('history-list');
  if (!history.length) { el.innerHTML='<div style="color:var(--muted);font-size:11px;text-align:center;padding:16px;">Chưa có lịch sử</div>'; return; }
  el.innerHTML = history.slice(0,8).map(h=>`
    <div class="hist-item">
      <span class="hist-emo-icon">${h.emoji}</span>
      <div class="hist-text">
        <div class="hist-main">"${h.text.slice(0,70)}${h.text.length>70?'...':''}"</div>
        <div class="hist-meta">${h.label} · ${h.conf} · ${h.time}</div>
      </div>
    </div>`).join('');
}

async function loadStats() {

    const r = await fetch(API+"/stats");
    const d = await r.json();
    document.getElementById('stat-feedback').textContent = d.feedback_count;
    document.getElementById('stat-alpha').textContent = (d.model_alpha||0.85).toFixed(2);
  } catch {}
}

}
